CREATE TABLE t_cash_type (
id bigint(20) unsigned NOT NULL AUTO_INCREMENT,
cash_type_code varchar(50) NOT NULL COMMENT '类型编码',
biz_type int(2) NOT NULL COMMENT '业务类型',
format_flag tinyint(1) unsigned DEFAULT '0' COMMENT '是否需要查找用户信息,0-不需要,1-需要',
second_type int(3) unsigned NOT NULL COMMENT '奖励类型',
profit_type int(3) NOT NULL DEFAULT '-1' COMMENT 'order中对应的profit类型',
scene varchar(100) NOT NULL COMMENT '类型描述',
descs varchar(100) NOT NULL COMMENT ' 使用类型时所展示文本',
free_flag tinyint(1) unsigned DEFAULT '0' COMMENT '该类型提现是否收费标志，0-需要，1-不需要',
create_time timestamp NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT '创建时间',
update_time timestamp NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP COMMENT '更新时间',
del_flag tinyint(1) unsigned NOT NULL DEFAULT '0' COMMENT '删除标志',
PRIMARY KEY (id)
) ENGINE=InnoDB AUTO_INCREMENT=267 DEFAULT CHARSET=utf8 COMMENT='奖励类型明细表';
